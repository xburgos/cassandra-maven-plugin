19  import java.io.BufferedReader;
20  import java.io.File;
21  import java.io.FileReader;
22  import java.io.IOException;
23  import java.util.ArrayList;
24  import java.util.Collection;

26  import java.util.HashMap;
27  import java.util.Map;

29  import org.codehaus.plexus.util.IOUtil;

31  /***
32   * Report for a file.
33   * 
34   * @author <a href="mailto:bellingard@gmail.com">Fabrice Bellingard </a>
35   */
36  public class FileReport
37      implements Comparable
38  {
39  
40      /***
41       * The file being analysed
42       */
43      private File file;
44  
45      /***
46       * The name of the class corresponding to this file
47       */
48      private String className;
49  
50      /***
51       * Pair values of a line number and a comment.
52       * Map of [Integer,String]  
53       */
54      private Map tagListing;
55  
56      /***
57       * Constructor
58       * @param className the complete name of analysed class 
59       * @param tagListing Map of the comments
60       */
61      public FileReport( File file )
62      {
63          this.file = file;
64          this.tagListing = new HashMap();
65      }
66  
67      /***
68       * Adds a new entry to the list of tags found for this file report
69       * @param comment
70       * @param lineIndex
71       */
72      public void addComment( String comment, int lineIndex )
73      {
74          tagListing.put( new Integer( lineIndex ), comment );
75      }
76  
77      /***
78       * Returns the path corresponding to the analysed class,
79       * for instance: org/apache/maven/plugins/taglist/beans/FileReport
80       * @return the file path
81       */
82      public String getClassNameWithSlash()
83      {
84          return className.replace( '.', '/' );
85      }
86  
87      /***
88       * Returns the complete name of the analysed class,
89       * for instance: org.apache.maven.plugin.taglist.beans.FileReport
90       * @return the full class name
91       */
92      public String getClassName()
93      {
94          if ( className != null )
95          {
96              return className;
97          }
98          // need to compute it (only once)
99          BufferedReader reader = null;
100         String packageName = null;
101         try
102         {
103             reader = new BufferedReader( new FileReader( file ) );
104             String currentLine = reader.readLine().trim();
105             while ( currentLine != null )
106             {
107                 if ( currentLine.startsWith( "package" ) )
108                 {
109                     packageName = currentLine.substring( 8 ).trim().replaceAll( ";", "" ).trim();
110                     break;
111                 }
112                 currentLine = reader.readLine().trim();
113             }
114         }
115         catch ( IOException e )
116         {
117             packageName = "unknown";
118         }
119         finally
120         {
121             IOUtil.close(reader);
122         }
123 
124         className = packageName + "." + file.getName().replaceAll( ".java", "" );
125 
126         return className;
127     }
128 
129     /***
130      * Returns the list of the comment line indexes
131      * @return Collection of Integer
132      */
133     public Collection getLineIndexes()
134     {
135         ArrayList list = new ArrayList();
136         list.addAll( tagListing.keySet() );
137         Collections.sort( list );
138         return list;
139     }
140 
141     /***
142      * Returns the comment for the corresponding line index
143      * @param lineIndex the index of the line
144      * @return the comment
145      */
146     public String getComment( Integer lineIndex )
147     {
148         return (String) tagListing.get( lineIndex );
149     }
150 
151     /*** 
152      * Cf. overriden method documentation.
153      * @see java.lang.Comparable#compareTo(java.lang.Object)
154      */
155     public int compareTo( Object o )
156     {
157         if ( o instanceof FileReport )
158         {
159             FileReport fileReport = (FileReport) o;
160             return this.getClassName().compareTo( fileReport.getClassName() );
161         }
162         else
163         {
164             return 0;
165         }
166     }

19  import java.io.File;
20  import java.util.Collection;
21  import java.util.HashMap;
22  import java.util.Iterator;
23  import java.util.Map;
24  
25  /***
26   * Report of the scan for a specific tag
27   * 
28   * @author <a href="mailto:bellingard@gmail.com">Fabrice Bellingard </a>
29   */
30  public class TagReport
31      implements Comparable
32  {
33  
34      /***
35       * Tag name
36       */
37      private String tagName;
38  
39      /***
40       * Map containing File objects as keys, and FileReport object
41       * as values.
42       */
43      private Map fileReportsMap;
44  
45      /***
46       * Number of tags found in the code.
47       */
48      private int tagCount;
49  
50      /***
51       * Constructor
52       * @param tagName
53       * @param fileReports
54       */
55      public TagReport( String tagName )
56      {
57          this.tagName = tagName;
58          this.fileReportsMap = new HashMap();
59          tagCount = -1;
60      }
61  
62      /***
63       * Returns the FileReport object corresponding to this file.
64       * If it does not exist yet, it will be created.
65       * @param file the file being analysed
66       * @return a FileReport object for this file
67       */
68      public FileReport getFileReport( File file )
69      {
70          Object report = fileReportsMap.get( file );
71          if ( report instanceof FileReport )
72          {
73              return (FileReport) report;
74          }
75          else
76          {
77              FileReport newFileReport = new FileReport( file );
78              fileReportsMap.put( file, newFileReport );
79              return newFileReport;
80          }
81      }
82  
83      /***
84       * Returns the collection of file reports for the tag.
85       * @return a Collection of FileReport objects.
86       */
87      public Collection getFileReports()
88      {
89          return fileReportsMap.values();
90      }
91  
92      /***
93       * Returns the name of the tag that was looked for.
94       * @return the name of the tag.
95       */
96      public String getTagName()
97      {
98          return tagName;
99      }
100 
101     /***
102      * Gives the number of comments found for that tag.
103      * @return the number of comments
104      */
105     public int getTagCount()
106     {
107         if ( tagCount > -1 )
108         {
109             return tagCount;
110         }
111         // tagCount was not computed yet
112         tagCount = 0;
113         for ( Iterator iter = fileReportsMap.values().iterator(); iter.hasNext(); )
114         {
115             FileReport fileReport = (FileReport) iter.next();
116             tagCount += fileReport.getLineIndexes().size();
117         }
118         return tagCount;
119     }
120 
121     /*** 
122      * Cf. overriden method documentation.
123      * @see java.lang.Comparable#compareTo(java.lang.Object)
124      */
125     public int compareTo( Object o )
126     {
127         if ( o instanceof TagReport )
128         {
129             TagReport tagReport = (TagReport) o;
130             return this.getTagName().compareTo( tagReport.getTagName() );
131         }
132         else
133         {
134             return 0;
135         }
136     }
137 
138 }
